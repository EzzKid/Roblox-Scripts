task.wait(2)
local Workspace = game:GetService("Workspace")

local part = Instance.new("Part")
part.Name = "Part"
part.Size = Vector3.new(50, 1, 50)
part.Anchored = true
part.Position = Vector3.new(-2700, -257, 320)
part.Parent = Workspace

task.wait(2)

-- Teleport destinations
local teleport_table = {
	location1 = Vector3.new(-2700, -254, 320)
}

-- Services
local TweenService = game:GetService("TweenService")
local tweenInfo = TweenInfo.new(2, Enum.EasingStyle.Linear)
local lp = game.Players.LocalPlayer

-- Teleport function with tween animation
local function bypass_teleport(v)
	if lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
		local hrp = lp.Character.HumanoidRootPart
		local goal = {CFrame = CFrame.new(v)}

		-- Create the tween
		local tween = TweenService:Create(hrp, tweenInfo, goal)

		tween:Play()
		tween.Completed:Wait() -- Wait until tween finishes
	end
end


while true do
	print("Teleporting to location1...")
	bypass_teleport(teleport_table.location1)
	print("Waiting 15 seconds before next teleport...")
	task.wait(25)
end
