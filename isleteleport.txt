task.wait(3) -- small delay to ensure character loads

local teleport_table = {
	location1 = Vector3.new(-4000, -262, -540)
}

-- Services
local tween_s = game:GetService("TweenService")
local tweeninfo = TweenInfo.new(4.5, Enum.EasingStyle.Circular)
local lp = game.Players.LocalPlayer

-- Teleport function with tween animation
function bypass_teleport(v)
	if lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
		local hrp = lp.Character.HumanoidRootPart
		local goal = {CFrame = CFrame.new(v)}

		-- Create the tween correctly
		local tween = tween_s:Create(hrp, tweeninfo, goal)

		print("Tweening to location:", v)
		tween:Play()
		tween.Completed:Wait() -- Wait until tween finishes
		print("Teleport complete!")
	end
end

print("Teleport to location1 successful")
bypass_teleport(teleport_table.location1)
